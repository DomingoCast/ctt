# ctt v1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `ctt`, a single-binary Zig CLI/TUI/MCP tool that tracks Claude-assisted development tasks across multiple git repos, derived from worktrees + PR + Linear state.

**Architecture:** Hexagonal. Pure `domain` and `application` modules with no I/O; `infra/inbound/{tui,cli,mcp}` and `infra/outbound/{sqlite,git,gh,linear}` adapters; `main.zig` is the only composition point. Ports are vtable interface structs (fat pointers). All async is sync + threads — no Zig async needed.

**Tech Stack:** Zig 0.14.x stable, libvaxis (TUI), zqlite (SQLite), zig-clap (CLI args), std.http.Client (Linear GraphQL), std.ChildProcess (git/gh shellouts), std.json (config + MCP), std.log.

**Spec:** `docs/superpowers/specs/2026-06-02-ctt-claude-task-tracker-design.md`

**Library APIs to verify before use:** Zig moves fast and libvaxis/zqlite APIs evolve. The code examples in this plan use the **shape** of the right call but exact function names/signatures must be checked against `build.zig.zon` versions resolved at task 0.1. When a tasks's example doesn't compile, consult `zig-cache/o/.../*.zig` for the actual lib source and adjust.

    const target   = b.standardTargetOptions(.{});
    const optimize = b.standardOptimizeOption(.{});

    const libvaxis = b.dependency("libvaxis", .{ .target = target, .optimize = optimize });

pub fn parse(
    allocator: std.mem.Allocator,
    branch: BranchName,
    patterns: []const ProviderPattern,
) !?TicketRef {
    for (patterns) |p| {
        if (try findMatch(allocator, branch.value, p)) |ref| return ref;
    }
    return null;
}

fn findMatch(allocator: std.mem.Allocator, s: []const u8, p: ProviderPattern) !?TicketRef {
    var i: usize = 0;
    while (i < s.len) {
        // skip non-alpha
        if (!std.ascii.isAlphabetic(s[i])) { i += 1; continue; }
        // try to read prefix
        const prefix_start = i;
        while (i < s.len and std.ascii.isAlphabetic(s[i])) : (i += 1) {}
        const prefix_len = i - prefix_start;
        if (prefix_len < p.prefix_min or prefix_len > p.prefix_max) continue;
        // require dash
        if (i >= s.len or s[i] != '-') continue;
        i += 1;
        // require digits
        const digits_start = i;
        while (i < s.len and std.ascii.isDigit(s[i])) : (i += 1) {}
        if (i == digits_start) continue;
        // boundary: end of string or non-alphanumeric
        if (i < s.len and (std.ascii.isAlphanumeric(s[i]))) continue;

        // build external_id = UPPER(prefix) ++ "-" ++ digits
        const out_len = prefix_len + 1 + (i - digits_start);
        var out = try allocator.alloc(u8, out_len);
        for (s[prefix_start .. prefix_start + prefix_len], 0..) |c, idx| out[idx] = std.ascii.toUpper(c);
        out[prefix_len] = '-';
        @memcpy(out[prefix_len + 1 ..], s[digits_start..i]);
        return TicketRef{ .provider = p.provider, .external_id = out };
    }
    return null;
}

pub fn parsePorcelain(a: std.mem.Allocator, text: []const u8) ![]d.WorktreeSnapshot {
    var out = std.ArrayList(d.WorktreeSnapshot).init(a);
    defer out.deinit();
    var it = std.mem.splitScalar(u8, text, '\n');
    var cur_path: ?[]const u8 = null;
    var cur_sha:  ?[]const u8 = null;
    var cur_branch: ?[]const u8 = null;
    while (it.next()) |line| {
        if (line.len == 0) {
            if (cur_path) |p| if (cur_sha) |s| if (cur_branch) |b| {
                try out.append(.{
                    .path = try a.dupe(u8, p),
                    .branch = .{ .value = try a.dupe(u8, b) },
                    .head_sha = .{ .value = try a.dupe(u8, s) },
                    .commits_ahead_of_default = 0,
                    .has_upstream = false,
                    .commits_ahead_of_upstream = null,
                });
            };
            cur_path = null; cur_sha = null; cur_branch = null;
            continue;
        }
        if (std.mem.startsWith(u8, line, "worktree ")) cur_path = line[9..];
        if (std.mem.startsWith(u8, line, "HEAD "))     cur_sha  = line[5..];
        if (std.mem.startsWith(u8, line, "branch "))   cur_branch = stripRefs(line[7..]);
    }
    // flush
    if (cur_path) |p| if (cur_sha) |s| if (cur_branch) |b| {
        try out.append(.{
            .path = try a.dupe(u8, p),
            .branch = .{ .value = try a.dupe(u8, b) },
            .head_sha = .{ .value = try a.dupe(u8, s) },
            .commits_ahead_of_default = 0, .has_upstream = false, .commits_ahead_of_upstream = null,
        });
    };
    return out.toOwnedSlice();
}

fn stripRefs(s: []const u8) []const u8 {
    if (std.mem.startsWith(u8, s, "refs/heads/")) return s[11..];
    return s;
}

pub fn readRequest(a: std.mem.Allocator, reader: anytype) !?Request {
    // MCP uses LSP-style headers OR newline-delimited JSON depending on transport;
    // for stdio we use newline-delimited.
    var line = std.ArrayList(u8).init(a);
    defer line.deinit();
    reader.streamUntilDelimiter(line.writer(), '\n', 1 * 1024 * 1024) catch |e| switch (e) {
        error.EndOfStream => return null,
        else => return e,
    };
    const parsed = try std.json.parseFromSlice(Request, a, line.items, .{ .ignore_unknown_fields = true });
    return parsed.value;
}

pub fn writeResponse(writer: anytype, id: std.json.Value, result: anytype) !void {
    try writer.writeByte('{');
    try writer.writeAll("\"jsonrpc\":\"2.0\",\"id\":");
    try std.json.stringify(id, .{}, writer);
    try writer.writeAll(",\"result\":");
    try std.json.stringify(result, .{}, writer);
    try writer.writeAll("}\n");
}

pub fn writeError(writer: anytype, id: std.json.Value, code: i32, message: []const u8) !void {
    try writer.print("{{\"jsonrpc\":\"2.0\",\"id\":", .{});
